    94тЖТ    94тЖТ/**
    95тЖТ    95тЖТ * Convert `?` placeholders in SQL to PostgreSQL `$1, $2, $3...` positional params.
    96тЖТ    96тЖТ */
    97тЖТ    97тЖТfunction convertPlaceholders(sql: string): string {
    98тЖТ    98тЖТ  let result = '';
    99тЖТ    99тЖТ  let paramIndex = 1;
   100тЖТ   100тЖТ  let inString = false;
   101тЖТ   101тЖТ  let stringChar = '';
   102тЖТ   102тЖТ  let i = 0;
   103тЖТ   103тЖТ
   104тЖТ   104тЖТ  while (i < sql.length) {
   105тЖТ   105тЖТ    const ch = sql[i];
   106тЖТ   106тЖТ    const next = sql[i + 1];
   107тЖТ   107тЖТ
   108тЖТ   108тЖТ    if (!inString && (ch === "'" || ch === '"')) {
   109тЖТ   109тЖТ      inString = true;
   110тЖТ   110тЖТ      stringChar = ch;
   111тЖТ   111тЖТ      result += ch;
   112тЖТ   112тЖТ      i++;
   113тЖТ   113тЖТ      continue;
   114тЖТ   114тЖТ    }
   115тЖТ   115тЖТ
   116тЖТ   116тЖТ    if (inString) {
   117тЖТ   117тЖТ      if (ch === stringChar) {
   118тЖТ   118тЖТ        if (next === stringChar) {
   119тЖТ   119тЖТ          result += ch + next;
   120тЖТ   120тЖТ          i += 2;
   121тЖТ   121тЖТ          continue;
   122тЖТ   122тЖТ        }
   123тЖТ   123тЖТ        inString = false;
   124тЖТ   124тЖТ      }
   125тЖТ   125тЖТ      result += ch;
   126тЖТ   126тЖТ      i++;
   127тЖТ   127тЖТ      continue;
   128тЖТ   128тЖТ    }
   129тЖТ   129тЖТ
   130тЖТ   130тЖТ    if (ch === '?' && next === '?') {
   131тЖТ   131тЖТ      result += '?';
   132тЖТ   132тЖТ      i += 2;
   133тЖТ   133тЖТ      continue;
   134тЖТ   134тЖТ    }
   135тЖТ   135тЖТ
   136тЖТ   136тЖТ    if (ch === '?') {
   137тЖТ   137тЖТ      result += `$${paramIndex}`;
   138тЖТ   138тЖТ      paramIndex++;
   139тЖТ   139тЖТ      i++;
   140тЖТ   140тЖТ      continue;
   141тЖТ   141тЖТ    }
   142тЖТ   142тЖТ
   143тЖТ   143тЖТ    result += ch;
   144тЖТ   144тЖТ    i++;
   145тЖТ   145тЖТ  }
   146тЖТ   146тЖТ
   147тЖТ   147тЖТ  return result;
   148тЖТ   148тЖТ}

   246тЖТ   246тЖТ * Convert sqlite_master queries to PostgreSQL information_schema equivalents.

   248тЖТ   248тЖТfunction convertSqliteMasterQuery(sql: string): string {
   249тЖТ   249тЖТ  const normalized = sql.trim();
   250тЖТ   250тЖТ
   251тЖТ   251тЖТ  const nameSqlInMatch = normalized.match(
   252тЖТ   252тЖТ    /SELECT\s+name\s*,\s*sql\s+FROM\s+sqlite_master\s+WHERE\s+type\s*=\s*'table'\s+AND\s+name\s+IN\s*\(([^)]+)\)/i
   253тЖТ   253тЖТ  );

   255тЖТ   255тЖТ    const tableNames = nameSqlInMatch[1]
   256тЖТ   256тЖТ      .split(',')
   257тЖТ   257тЖТ      .map(t => t.trim().replace(/'/g, ''))
   258тЖТ   258тЖТ      .map(t => `'${t}'`)
   259тЖТ   259тЖТ      .join(', ');
   260тЖТ   260тЖТ    return `SELECT table_name AS name, 'See pg_catalog for DDL' AS sql FROM information_schema.tables WHERE table_schema = 'public' AND table_name IN (${tableNames})`;

   262тЖТ   262тЖТ
   263тЖТ   263тЖТ  const sqlNameMatch = normalized.match(
   264тЖТ   264тЖТ    /SELECT\s+sql\s+FROM\s+sqlite_master\s+WHERE\s+type\s*=\s*'table'\s+AND\s+name\s*=\s*\?/i

   266тЖТ   266тЖТ  if (sqlNameMatch) {
   267тЖТ   267тЖТ    return `SELECT 'See pg_catalog for DDL' AS sql FROM information_schema.tables WHERE table_schema = 'public' AND table_name = $1`;
   268тЖТ   268тЖТ  }

   270тЖТ   270тЖТ  const indexMatch = normalized.match(
   271тЖТ   271тЖТ    /SELECT\s+name\s*,\s*sql\s+FROM\s+sqlite_master\s+WHERE\s+type\s*=\s*'index'\s+AND\s+tbl_name\s*=\s*\?/i

   282тЖТ   282тЖТ  }
   283тЖТ   283тЖТ
   284тЖТ   284тЖТ  const notLikeMatch = normalized.match(
   285тЖТ   285тЖТ    /SELECT\s+name\s+FROM\s+sqlite_master\s+WHERE\s+type\s*=\s*'table'\s+AND\s+name\s+NOT\s+LIKE\s+'sqlite_%'(?:\s+AND\s+name\s*!=\s*'([^']+)')?(?:\s+ORDER\s+BY\s+name)?/i

   299тЖТ   299тЖТ  if (allTablesMatch) {
   300тЖТ   300тЖТ    return `SELECT table_name AS name FROM information_schema.tables WHERE table_schema = 'public' AND table_type = 'BASE TABLE' ORDER BY table_name`;
   301тЖТ   301тЖТ  }

   303тЖТ   303тЖТ  const nameInMatch = normalized.match(

   313тЖТ   313тЖТ  }
   314тЖТ   314тЖТ
   315тЖТ   315тЖТ  console.warn(`[DB-POSTGRES] Unhandled sqlite_master query: ${sql.substring(0, 100)}`);
   316тЖТ   316тЖТ  return `SELECT table_name AS name FROM information_schema.tables WHERE table_schema = 'public'`;
   317тЖТ   317тЖТ}

   371тЖТ   371тЖТ/**
   372тЖТ   372тЖТ * Extract table name from SQL for enum type mapping.
   373тЖТ   373тЖТ * Works with both quoted and unquoted table names.
   374тЖТ   374тЖТ */
   375тЖТ   375тЖТfunction extractTableName(sql: string): string | null {
   376тЖТ   376тЖТ  // UPDATE "TableName" SET ...  or  UPDATE TableName SET ...
   377тЖТ   377тЖТ  const updateMatch = sql.match(/UPDATE\s+"?(\w+)"?\s+SET/i);
   378тЖТ   378тЖТ  if (updateMatch) return updateMatch[1];
   379тЖТ   379тЖТ  // INSERT INTO "TableName" (...)  or  INSERT INTO TableName (...)
   380тЖТ   380тЖТ  const insertMatch = sql.match(/INSERT\s+INTO\s+"?(\w+)"?\s*\(/i);
   381тЖТ   381тЖТ  if (insertMatch) return insertMatch[1];
   382тЖТ   382тЖТ  return null;
   383тЖТ   383тЖТ}
   384тЖТ   384тЖТ
   385тЖТ   385тЖТ/**
   386тЖТ   386тЖТ * Convert inline boolean literals in SQL for PostgreSQL compatibility.
   387тЖТ   387тЖТ * Replaces `booleanCol = 0` тЖТ `booleanCol = false`, `booleanCol = 1` тЖТ `booleanCol = true`
   388тЖТ   388тЖТ * Also handles SET clause: `SET booleanCol = 1` тЖТ `SET booleanCol = true`
   389тЖТ   389тЖТ * Also handles INSERT VALUES: inline 0/1 for boolean columns тЖТ false/true
   390тЖТ   390тЖТ */
   391тЖТ   391тЖТfunction convertInlineBooleanLiterals(sql: string): string {
   392тЖТ   392тЖТ  // Pattern 1: WHERE/SET clause тАФ `booleanCol = 0` or `booleanCol = 1`
   393тЖТ   393тЖТ  // Must handle: `isRead = 0`, `isRead = 1`, `SET isRead = 1`, `AND isRead = 0`
   394тЖТ   394тЖТ  let result = sql;
   395тЖТ   395тЖТ
   396тЖТ   396тЖТ  // Replace `booleanCol = 0` тЖТ `booleanCol = false` (only outside of strings)
   397тЖТ   397тЖТ  result = result.replace(
   398тЖТ   398тЖТ    new RegExp(`\\b(${BOOLEAN_COLUMNS_PATTERN})\\s*=\\s*0\\b`, 'gi'),
   399тЖТ   399тЖТ    '$1 = false'
   400тЖТ   400тЖТ  );
   401тЖТ   401тЖТ
   402тЖТ   402тЖТ  // Replace `booleanCol = 1` тЖТ `booleanCol = true` (only outside of strings)
   403тЖТ   403тЖТ  result = result.replace(
   404тЖТ   404тЖТ    new RegExp(`\\b(${BOOLEAN_COLUMNS_PATTERN})\\s*=\\s*1\\b`, 'gi'),
   405тЖТ   405тЖТ    '$1 = true'
   406тЖТ   406тЖТ  );
   407тЖТ   407тЖТ
   408тЖТ   408тЖТ  // Pattern 2: INSERT VALUES тАФ convert inline 0/1 for boolean columns in VALUES clause
   409тЖТ   409тЖТ  // e.g. INSERT INTO Notification (id, userId, isRead, emailSent, ...) VALUES (?, ?, 0, 0, ...)
   410тЖТ   410тЖТ  //                                                                                      тЖС    тЖС  тЖТ false, false
   411тЖТ   411тЖТ  result = convertInsertBooleanValues(result);
   412тЖТ   412тЖТ
   413тЖТ   413тЖТ  return result;
   414тЖТ   414тЖТ}
   415тЖТ   415тЖТ
   416тЖТ   416тЖТ/**
   417тЖТ   417тЖТ * Parse INSERT statements and convert inline 0/1 literals in the VALUES clause
   418тЖТ   418тЖТ * to PostgreSQL-compatible false/true for boolean column positions.
   419тЖТ   419тЖТ *
   420тЖТ   420тЖТ * Example transformation:
   421тЖТ   421тЖТ *   INSERT INTO Notification (id, isRead, emailSent) VALUES (?, 0, 0)
   422тЖТ   422тЖТ *   тЖТ INSERT INTO Notification (id, isRead, emailSent) VALUES (?, false, false)
   423тЖТ   423тЖТ */
   424тЖТ   424тЖТfunction convertInsertBooleanValues(sql: string): string {
   425тЖТ   425тЖТ  // Only process INSERT statements
   426тЖТ   426тЖТ  const insertMatch = sql.match(
   427тЖТ   427тЖТ    /^(INSERT\s+INTO\s+(\S+)\s*\()([^)]+)(\)\s*VALUES\s*\()([\s\S]*?)(\);?\s*$)/im
   428тЖТ   428тЖТ  );
   429тЖТ   429тЖТ  if (!insertMatch) return sql;
   430тЖТ   430тЖТ
   431тЖТ   431тЖТ  const [, prefix, _tableName, columnsStr, middle, valuesStr, suffix] = insertMatch;
   432тЖТ   432тЖТ
   433тЖТ   433тЖТ  // Parse column names тАФ split by comma and clean up whitespace/quotes
   434тЖТ   434тЖТ  const columns = columnsStr.split(',').map(c =>
   435тЖТ   435тЖТ    c.trim().replace(/^"(.*)"$/, '$1')
   436тЖТ   436тЖТ  );
   437тЖТ   437тЖТ
   438тЖТ   438тЖТ  // Find positions of boolean columns (0-indexed)
   439тЖТ   439тЖТ  const boolColSet = new Set<string>(BOOLEAN_COLUMNS.map(c => c.toLowerCase()));
   440тЖТ   440тЖТ  const boolPositions = new Set<number>();
   441тЖТ   441тЖТ  columns.forEach((col, idx) => {
   442тЖТ   442тЖТ    if (boolColSet.has(col.toLowerCase())) {
   443тЖТ   443тЖТ      boolPositions.add(idx);
   444тЖТ   444тЖТ    }
   445тЖТ   445тЖТ  });
   446тЖТ   446тЖТ
   447тЖТ   447тЖТ  if (boolPositions.size === 0) return sql;
   448тЖТ   448тЖТ
   449тЖТ   449тЖТ  // Parse the VALUES clause тАФ simple tokenizer that respects strings and nested parens
   450тЖТ   450тЖТ  const values: string[] = [];
   451тЖТ   451тЖТ  let currentValue = '';
   452тЖТ   452тЖТ  let inString = false;
   453тЖТ   453тЖТ  let stringChar = '';
   454тЖТ   454тЖТ  let depth = 0;
   455тЖТ   455тЖТ
   456тЖТ   456тЖТ  for (let i = 0; i < valuesStr.length; i++) {
   457тЖТ   457тЖТ    const ch = valuesStr[i];
   458тЖТ   458тЖТ
   459тЖТ   459тЖТ    if (!inString && (ch === "'" || ch === '"')) {
   460тЖТ   460тЖТ      inString = true;
   461тЖТ   461тЖТ      stringChar = ch;
   462тЖТ   462тЖТ      currentValue += ch;
   463тЖТ   463тЖТ      continue;
   464тЖТ   464тЖТ    }
   465тЖТ   465тЖТ
   466тЖТ   466тЖТ    if (inString) {
   467тЖТ   467тЖТ      currentValue += ch;
   468тЖТ   468тЖТ      if (ch === stringChar) {
   469тЖТ   469тЖТ        if (valuesStr[i + 1] === stringChar) {
   470тЖТ   470тЖТ          // Escaped quote (e.g., 'it''s')
   471тЖТ   471тЖТ          currentValue += valuesStr[i + 1];
   472тЖТ   472тЖТ          i++;
   473тЖТ   473тЖТ          continue;
   474тЖТ   474тЖТ        }
   475тЖТ   475тЖТ        inString = false;
   476тЖТ   476тЖТ      }
   477тЖТ   477тЖТ      continue;
   478тЖТ   478тЖТ    }
   479тЖТ   479тЖТ
   480тЖТ   480тЖТ    if (ch === '(') { depth++; currentValue += ch; continue; }
   481тЖТ   481тЖТ    if (ch === ')') { depth--; currentValue += ch; continue; }
   482тЖТ   482тЖТ
   483тЖТ   483тЖТ    if (ch === ',' && depth === 0) {
   484тЖТ   484тЖТ      values.push(currentValue.trim());
   485тЖТ   485тЖТ      currentValue = '';
   486тЖТ   486тЖТ      continue;
   487тЖТ   487тЖТ    }
   488тЖТ   488тЖТ
   489тЖТ   489тЖТ    currentValue += ch;
   490тЖТ   490тЖТ  }
   491тЖТ   491тЖТ  if (currentValue.trim()) values.push(currentValue.trim());
   492тЖТ   492тЖТ
   493тЖТ   493тЖТ  // Replace 0/1 at boolean column positions with false/true
   494тЖТ   494тЖТ  let changed = false;
   495тЖТ   495тЖТ  for (let i = 0; i < columns.length && i < values.length; i++) {
   496тЖТ   496тЖТ    if (boolPositions.has(i)) {
   497тЖТ   497тЖТ      if (values[i] === '0') { values[i] = 'false'; changed = true; }
   498тЖТ   498тЖТ      else if (values[i] === '1') { values[i] = 'true'; changed = true; }
   499тЖТ   499тЖТ    }
   500тЖТ   500тЖТ  }
   501тЖТ   501тЖТ
   502тЖТ   502тЖТ  if (!changed) return sql;
   503тЖТ   503тЖТ
   504тЖТ   504тЖТ  return `${prefix}${columnsStr}${middle}${values.join(', ')}${suffix}`;
   505тЖТ   505тЖТ}

   507тЖТ   507тЖТ/**
   508тЖТ   508тЖТ * Convert SQLite-specific SQL syntax to PostgreSQL equivalents.
   509тЖТ   509тЖТ */
   510тЖТ   510тЖТfunction convertSqlDialect(sql: string): string {
   511тЖТ   511тЖТ  let converted = sql.trim();
   512тЖТ   512тЖТ
   513тЖТ   513тЖТ  if (/^PRAGMA\s/i.test(converted)) {
   514тЖТ   514тЖТ    return '';
   515тЖТ   515тЖТ  }
   516тЖТ   516тЖТ
   517тЖТ   517тЖТ  if (/sqlite_master/i.test(converted)) {
   518тЖТ   518тЖТ    converted = convertSqliteMasterQuery(converted);
   519тЖТ   519тЖТ  }
   520тЖТ   520тЖТ
   521тЖТ   521тЖТ  // Convert inline boolean literals BEFORE placeholder conversion
   522тЖТ   522тЖТ  // so the column names are still in their original form
   523тЖТ   523тЖТ  converted = convertInlineBooleanLiterals(converted);
   524тЖТ   524тЖТ
   525тЖТ   525тЖТ  const wasInsertOrIgnore = /^INSERT\s+OR\s+IGNORE\s+/i.test(converted);
   526тЖТ   526тЖТ  if (wasInsertOrIgnore) {
   527тЖТ   527тЖТ    converted = converted.replace(/^INSERT\s+OR\s+IGNORE\s+/i, 'INSERT ');
   528тЖТ   528тЖТ  }
   529тЖТ   529тЖТ
   530тЖТ   530тЖТ  const wasInsertOrReplace = /^INSERT\s+OR\s+REPLACE\s+/i.test(converted);
   531тЖТ   531тЖТ  if (wasInsertOrReplace) {
   532тЖТ   532тЖТ    converted = converted.replace(/^INSERT\s+OR\s+REPLACE\s+/i, 'INSERT ');
   533тЖТ   533тЖТ  }
   534тЖТ   534тЖТ
   535тЖТ   535тЖТ  converted = convertPlaceholders(converted);
   536тЖТ   536тЖТ
   537тЖТ   537тЖТ  if (wasInsertOrIgnore) {
   538тЖТ   538тЖТ    const returningIdx = converted.toUpperCase().indexOf('RETURNING');
   539тЖТ   539тЖТ    if (returningIdx !== -1) {
   540тЖТ   540тЖТ      converted = `${converted.substring(0, returningIdx)} ON CONFLICT DO NOTHING${converted.substring(returningIdx)}`;
   541тЖТ   541тЖТ    } else {
   542тЖТ   542тЖТ      converted = `${converted} ON CONFLICT DO NOTHING`;
   543тЖТ   543тЖТ    }
   544тЖТ   544тЖТ  }
   545тЖТ   545тЖТ
   546тЖТ   546тЖТ  if (wasInsertOrReplace) {
   547тЖТ   547тЖТ    const insertMatch = converted.match(/^INSERT\s+INTO\s+"?(\w+)"?\s*\(([^)]+)\)\s*VALUES\s*\(/i);
   548тЖТ   548тЖТ    if (insertMatch) {
   549тЖТ   549тЖТ      // Extract column names from INSERT to build proper SET clause
   550тЖТ   550тЖТ      const columns = insertMatch[2].split(',').map(c =>
   551тЖТ   551тЖТ        c.trim().replace(/^"(.*)"$/, '$1')
   552тЖТ   552тЖТ      );
   553тЖТ   553тЖТ      // Update all columns except 'id' (the conflict target) for proper upsert
   554тЖТ   554тЖТ      const setClause = columns
   555тЖТ   555тЖТ        .filter(col => col !== 'id')
   556тЖТ   556тЖТ        .map(col => `"${col}" = EXCLUDED."${col}"`)
   557тЖТ   557тЖТ        .join(', ');
   558тЖТ   558тЖТ      const upsertClause = setClause
   559тЖТ   559тЖТ        ? `ON CONFLICT (id) DO UPDATE SET ${setClause}`
   560тЖТ   560тЖТ        : `ON CONFLICT (id) DO UPDATE SET id = EXCLUDED.id`;
   561тЖТ   561тЖТ
   562тЖТ   562тЖТ      const returningIdx = converted.toUpperCase().indexOf('RETURNING');
   563тЖТ   563тЖТ      if (returningIdx !== -1) {
   564тЖТ   564тЖТ        converted = `${converted.substring(0, returningIdx)} ${upsertClause}${converted.substring(returningIdx)}`;
   565тЖТ   565тЖТ      } else {
   566тЖТ   566тЖТ        converted = `${converted} ${upsertClause}`;
   567тЖТ   567тЖТ      }
   568тЖТ   568тЖТ    }
   569тЖТ   569тЖТ  }
   570тЖТ   570тЖТ
   571тЖТ   571тЖТ  converted = converted.replace(/\bIFNULL\s*\(/gi, 'COALESCE(');
   572тЖТ   572тЖТ
   573тЖТ   573тЖТ  converted = converted.replace(
   574тЖТ   574тЖТ    /\bGROUP_CONCAT\s*\(\s*([^,)]+)\s*,\s*([^)]+)\s*\)/gi,
   575тЖТ   575тЖТ    'STRING_AGG($1::text, $2)'
   576тЖТ   576тЖТ  );
   577тЖТ   577тЖТ  converted = converted.replace(
   578тЖТ   578тЖТ    /\bGROUP_CONCAT\s*\(\s*([^)]+)\s*\)/gi,
   579тЖТ   579тЖТ    'STRING_AGG($1::text, \'\,\')'
   580тЖТ   580тЖТ  );
   581тЖТ   581тЖТ
   582тЖТ   582тЖТ  converted = quoteUppercaseIdentifiers(converted);
   583тЖТ   583тЖТ
   584тЖТ   584тЖТ  // Add ::text cast to enum column comparisons so PostgreSQL can compare
   585тЖТ   585тЖТ  // enum types with text parameters (e.g., status = $1 where status is DemandStatus)
   586тЖТ   586тЖТ  // After quoteUppercaseIdentifiers, enum columns may be quoted like "status"
   587тЖТ   587тЖТ  // Also adds ::"EnumType" casts to parameters in SET context
   588тЖТ   588тЖТ  converted = addEnumTextCasts(converted);
   589тЖТ   589тЖТ
   590тЖТ   590тЖТ  // Add enum type casts to parameters in INSERT VALUES clauses
   591тЖТ   591тЖТ  // (PostgreSQL cannot auto-cast text params to enum in VALUES)
   592тЖТ   592тЖТ  converted = addEnumInsertCasts(converted);
   593тЖТ   593тЖТ
   594тЖТ   594тЖТ  return converted;
   595тЖТ   595тЖТ}
   596тЖТ   596тЖТ
   597тЖТ   597тЖТ/**
   598тЖТ   598тЖТ * Add ::text cast to enum column comparisons AND enum type casts to SET parameters.
   599тЖТ   599тЖТ *
   600тЖТ   600тЖТ * Two transformations:
   601тЖТ   601тЖТ *
   602тЖТ   602тЖТ * 1. WHERE context тАФ cast enum column to text for comparison with text params:
   603тЖТ   603тЖТ *   "status" = $1          тЖТ  "status"::text = $1
   604тЖТ   604тЖТ *   "status" = 'ACTIVE'    тЖТ  "status"::text = 'ACTIVE'
   605тЖТ   605тЖТ *
   606тЖТ   606тЖТ * 2. SET context тАФ cast text parameter to the correct enum type:
   607тЖТ   607тЖТ *   SET "visibilityPackage" = $1  тЖТ  SET "visibilityPackage" = $1::"VisibilityPackage"
   608тЖТ   608тЖТ *   SET "status" = $1            тЖТ  SET "status" = $1::"DemandStatus"
   609тЖТ   609тЖТ *
   610тЖТ   610тЖТ * PostgreSQL enum types cannot be compared directly with text, and also
   611тЖТ   611тЖТ * cannot auto-cast text parameters TO enum types in SET/VALUES clauses.
   612тЖТ   612тЖТ */
   613тЖТ   613тЖТfunction addEnumTextCasts(sql: string): string {
   614тЖТ   614тЖТ  let result = sql;
   615тЖТ   615тЖТ  let output = '';
   616тЖТ   616тЖТ  let i = 0;
   617тЖТ   617тЖТ  const len = result.length;
   618тЖТ   618тЖТ  let inSingleQuote = false;
   619тЖТ   619тЖТ  let inDoubleQuote = false;
   620тЖТ   620тЖТ
   621тЖТ   621тЖТ  // Track assignment vs comparison context
   622тЖТ   622тЖТ  let inAssignmentContext = false;
   623тЖТ   623тЖТ
   624тЖТ   624тЖТ  // Track pending enum type cast for the next $N parameter in SET context
   625тЖТ   625тЖТ  let pendingEnumCast: string | null = null;
   626тЖТ   626тЖТ
   627тЖТ   627тЖТ  // Extract table name for enum type lookup
   628тЖТ   628тЖТ  const tableName = extractTableName(sql);
   629тЖТ   629тЖТ
   630тЖТ   630тЖТ  // Keywords that END the SET assignment context
   631тЖТ   631тЖТ  const assignmentEndKeywords = ['WHERE', 'RETURNING', 'FROM', 'LIMIT', 'ORDER', 'GROUP', 'HAVING', 'WINDOW', 'FETCH', 'OFFSET'];
   632тЖТ   632тЖТ  // Keywords that START the SET assignment context
   633тЖТ   633тЖТ  const assignmentStartKeywords = ['SET'];
   634тЖТ   634тЖТ
   635тЖТ   635тЖТ  while (i < len) {
   636тЖТ   636тЖТ    const ch = result[i];
   637тЖТ   637тЖТ
   638тЖТ   638тЖТ    // Track string states
   639тЖТ   639тЖТ    if (!inDoubleQuote && ch === "'") {
   640тЖТ   640тЖТ      if (inSingleQuote) {
   641тЖТ   641тЖТ        if (result[i + 1] === "'") {
   642тЖТ   642тЖТ          output += "''";
   643тЖТ   643тЖТ          i += 2;
   644тЖТ   644тЖТ          continue;
   645тЖТ   645тЖТ        }
   646тЖТ   646тЖТ        inSingleQuote = false;
   647тЖТ   647тЖТ      } else {
   648тЖТ   648тЖТ        inSingleQuote = true;
   649тЖТ   649тЖТ      }
   650тЖТ   650тЖТ      output += ch;
   651тЖТ   651тЖТ      i++;
   652тЖТ   652тЖТ      continue;
   653тЖТ   653тЖТ    }
   654тЖТ   654тЖТ
   655тЖТ   655тЖТ    if (!inSingleQuote && ch === '"') {
   656тЖТ   656тЖТ      inDoubleQuote = !inDoubleQuote;
   657тЖТ   657тЖТ      output += ch;
   658тЖТ   658тЖТ      i++;
   659тЖТ   659тЖТ      continue;
   660тЖТ   660тЖТ    }
   661тЖТ   661тЖТ
   662тЖТ   662тЖТ    if (inSingleQuote || inDoubleQuote) {
   663тЖТ   663тЖТ      output += ch;
   664тЖТ   664тЖТ      i++;
   665тЖТ   665тЖТ      continue;
   666тЖТ   666тЖТ    }
   667тЖТ   667тЖТ
   668тЖТ   668тЖТ    // Handle $N parameters тАФ add enum type cast if pending
   669тЖТ   669тЖТ    if (ch === '$' && /\d/.test(result[i + 1] || '')) {
   670тЖТ   670тЖТ      let paramNum = '';
   671тЖТ   671тЖТ      let j = i + 1;
   672тЖТ   672тЖТ      while (j < len && /\d/.test(result[j])) {
   673тЖТ   673тЖТ        paramNum += result[j];
   674тЖТ   674тЖТ        j++;
   675тЖТ   675тЖТ      }
   676тЖТ   676тЖТ
   677тЖТ   677тЖТ      if (pendingEnumCast) {
   678тЖТ   678тЖТ        output += `$${paramNum}::"${pendingEnumCast}"`;
   679тЖТ   679тЖТ        pendingEnumCast = null;
   680тЖТ   680тЖТ      } else {
   681тЖТ   681тЖТ        output += `$${paramNum}`;
   682тЖТ   682тЖТ      }
   683тЖТ   683тЖТ      i = j;
   684тЖТ   684тЖТ      continue;
   685тЖТ   685тЖТ    }
   686тЖТ   686тЖТ
   687тЖТ   687тЖТ    // Detect SQL keywords to track assignment vs comparison context
   688тЖТ   688тЖТ    if (/[a-zA-Z]/.test(ch) && (i === 0 || !/[a-zA-Z0-9_]/.test(result[i - 1]))) {
   689тЖТ   689тЖТ      let word = '';
   690тЖТ   690тЖТ      let j = i;
   691тЖТ   691тЖТ      while (j < len && /[a-zA-Z0-9_]/.test(result[j])) {
   692тЖТ   692тЖТ        word += result[j];
   693тЖТ   693тЖТ        j++;
   694тЖТ   694тЖТ      }
   695тЖТ   695тЖТ      const upperWord = word.toUpperCase();
   696тЖТ   696тЖТ
   697тЖТ   697тЖТ      if (assignmentStartKeywords.includes(upperWord) && upperWord === 'SET') {
   698тЖТ   698тЖТ        inAssignmentContext = true;
   699тЖТ   699тЖТ      } else if (assignmentEndKeywords.some(kw => kw === upperWord)) {
   700тЖТ   700тЖТ        inAssignmentContext = false;
   701тЖТ   701тЖТ        pendingEnumCast = null; // Reset if we leave SET context
   702тЖТ   702тЖТ      }
   703тЖТ   703тЖТ
   704тЖТ   704тЖТ      // If we have a pending enum cast and encounter a non-parameter word
   705тЖТ   705тЖТ      // (like EXCLUDED, CURRENT_TIMESTAMP, NULL, etc.), clear the pending cast
   706тЖТ   706тЖТ      // since the value is not a $N parameter that needs casting
   707тЖТ   707тЖТ      if (pendingEnumCast && !upperWord.startsWith('$')) {
   708тЖТ   708тЖТ        pendingEnumCast = null;
   709тЖТ   709тЖТ      }
   710тЖТ   710тЖТ      // Don't consume the word тАФ let it fall through to enum column check
   711тЖТ   711тЖТ    }
   712тЖТ   712тЖТ
   713тЖТ   713тЖТ    // Check if we're at an enum column name (quoted or unquoted)
   714тЖТ   714тЖТ    let matched = false;
   715тЖТ   715тЖТ
   716тЖТ   716тЖТ    for (const enumCol of ENUM_COLUMNS) {
   717тЖТ   717тЖТ      // Check quoted form: "enumCol"
   718тЖТ   718тЖТ      const quotedPattern = `"${enumCol}"`;
   719тЖТ   719тЖТ      if (result.substring(i, i + quotedPattern.length) === quotedPattern) {
   720тЖТ   720тЖТ        const afterCol = result.substring(i + quotedPattern.length).trimStart();
   721тЖТ   721тЖТ        const nextNonSpace = afterCol[0];
   722тЖТ   722тЖТ
   723тЖТ   723тЖТ        // Skip if already has ::text
   724тЖТ   724тЖТ        if (afterCol.startsWith('::text')) {
   725тЖТ   725тЖТ          output += quotedPattern;
   726тЖТ   726тЖТ          i += quotedPattern.length;
   727тЖТ   727тЖТ          matched = true;
   728тЖТ   728тЖТ          break;
   729тЖТ   729тЖТ        }
   730тЖТ   730тЖТ
   731тЖТ   731тЖТ        // Check if this is a comparison/assignment context (=, !=, <>, IN, NOT IN)
   732тЖТ   732тЖТ        if (nextNonSpace === '=' || afterCol.startsWith('!=') || afterCol.startsWith('<>') ||
   733тЖТ   733тЖТ            afterCol.match(/^(NOT\s+)?IN\s*\(/i)) {
   734тЖТ   734тЖТ          if (!inAssignmentContext) {
   735тЖТ   735тЖТ            // WHERE context: add ::text to column for text comparison
   736тЖТ   736тЖТ            output += `${quotedPattern}::text`;
   737тЖТ   737тЖТ          } else {
   738тЖТ   738тЖТ            // SET context: don't add ::text, but set pending cast for the next $N param
   739тЖТ   739тЖТ            const enumType = tableName ? (ENUM_TYPE_MAP[`${tableName}.${enumCol}`] || null) : null;
   740тЖТ   740тЖТ            if (enumType) {
   741тЖТ   741тЖТ              pendingEnumCast = enumType;
   742тЖТ   742тЖТ            }
   743тЖТ   743тЖТ            output += quotedPattern;
   744тЖТ   744тЖТ          }
   745тЖТ   745тЖТ          i += quotedPattern.length;
   746тЖТ   746тЖТ          matched = true;
   747тЖТ   747тЖТ          break;
   748тЖТ   748тЖТ        }
   749тЖТ   749тЖТ      }
   750тЖТ   750тЖТ
   751тЖТ   751тЖТ      // Check unquoted form: enumCol (lowercase, no uppercase letters so no quotes)
   752тЖТ   752тЖТ      const unquotedPattern = enumCol;
   753тЖТ   753тЖТ      if (result.substring(i, i + unquotedPattern.length) === unquotedPattern) {
   754тЖТ   754тЖТ        // Ensure word boundary before
   755тЖТ   755тЖТ        if (i > 0 && /[a-zA-Z0-9_]/.test(result[i - 1])) continue;
   756тЖТ   756тЖТ        // Ensure word boundary after (not followed by alphanumeric)
   757тЖТ   757тЖТ        const afterMatch = result[i + unquotedPattern.length];
   758тЖТ   758тЖТ        if (afterMatch && /[a-zA-Z0-9_]/.test(afterMatch)) continue;
   759тЖТ   759тЖТ
   760тЖТ   760тЖТ        const afterCol = result.substring(i + unquotedPattern.length).trimStart();
   761тЖТ   761тЖТ        const nextNonSpace = afterCol[0];
   762тЖТ   762тЖТ
   763тЖТ   763тЖТ        // Skip if already has ::text
   764тЖТ   764тЖТ        if (afterCol.startsWith('::text')) {
   765тЖТ   765тЖТ          output += unquotedPattern;
   766тЖТ   766тЖТ          i += unquotedPattern.length;
   767тЖТ   767тЖТ          matched = true;
   768тЖТ   768тЖТ          break;
   769тЖТ   769тЖТ        }
   770тЖТ   770тЖТ
   771тЖТ   771тЖТ        // Check if this is a comparison/assignment context
   772тЖТ   772тЖТ        if (nextNonSpace === '=' || afterCol.startsWith('!=') || afterCol.startsWith('<>') ||
   773тЖТ   773тЖТ            afterCol.match(/^(NOT\s+)?IN\s*\(/i)) {
   774тЖТ   774тЖТ          if (!inAssignmentContext) {
   775тЖТ   775тЖТ            // WHERE context: add ::text to column for text comparison
   776тЖТ   776тЖТ            output += `${unquotedPattern}::text`;
   777тЖТ   777тЖТ          } else {
   778тЖТ   778тЖТ            // SET context: don't add ::text, but set pending cast for the next $N param
   779тЖТ   779тЖТ            const enumType = tableName ? (ENUM_TYPE_MAP[`${tableName}.${enumCol}`] || null) : null;
   780тЖТ   780тЖТ            if (enumType) {
   781тЖТ   781тЖТ              pendingEnumCast = enumType;
   782тЖТ   782тЖТ            }
   783тЖТ   783тЖТ            output += unquotedPattern;
   784тЖТ   784тЖТ          }
   785тЖТ   785тЖТ          i += unquotedPattern.length;
   786тЖТ   786тЖТ          matched = true;
   787тЖТ   787тЖТ          break;
   788тЖТ   788тЖТ        }
   789тЖТ   789тЖТ      }
   790тЖТ   790тЖТ    }
   791тЖТ   791тЖТ
   792тЖТ   792тЖТ    if (!matched) {
   793тЖТ   793тЖТ      output += ch;
   794тЖТ   794тЖТ      i++;
   795тЖТ   795тЖТ    }
   796тЖТ   796тЖТ  }
   797тЖТ   797тЖТ
   798тЖТ   798тЖТ  return output;
   799тЖТ   799тЖТ}

   801тЖТ   801тЖТ/**
   802тЖТ   802тЖТ * Add enum type casts to parameters in INSERT VALUES clauses.
   803тЖТ   803тЖТ *
   804тЖТ   804тЖТ * PostgreSQL cannot auto-cast text parameters to enum types in VALUES.
   805тЖТ   805тЖТ * For example:
   806тЖТ   806тЖТ *   INSERT INTO "Demand" (..., "visibilityPackage", ...) VALUES (..., $5, ...)
   807тЖТ   807тЖТ *   тЖТ INSERT INTO "Demand" (..., "visibilityPackage", ...) VALUES (..., $5::"VisibilityPackage", ...)
   808тЖТ   808тЖТ *
   809тЖТ   809тЖТ * This function parses the column list, identifies enum column positions,
   810тЖТ   810тЖТ * and adds explicit casts to the corresponding parameters.
   811тЖТ   811тЖТ */
   812тЖТ   812тЖТfunction addEnumInsertCasts(sql: string): string {
   813тЖТ   813тЖТ  // Only process INSERT statements
   814тЖТ   814тЖТ  const insertMatch = sql.match(
   815тЖТ   815тЖТ    /^(INSERT\s+INTO\s+"?(\w+)"?\s*\()([^)]+)(\)\s*VALUES\s*\()([\s\S]*?)(\);?\s*$)/im
   816тЖТ   816тЖТ  );
   817тЖТ   817тЖТ  if (!insertMatch) return sql;
   818тЖТ   818тЖТ
   819тЖТ   819тЖТ  const [, prefix, tableName, columnsStr, middle, valuesStr, suffix] = insertMatch;
   820тЖТ   820тЖТ
   821тЖТ   821тЖТ  // Parse column names
   822тЖТ   822тЖТ  const columns = columnsStr.split(',').map(c =>
   823тЖТ   823тЖТ    c.trim().replace(/^"(.*)"$/, '$1')
   824тЖТ   824тЖТ  );
   825тЖТ   825тЖТ
   826тЖТ   826тЖТ  // Find positions of enum columns that have a mapping for this table
   827тЖТ   827тЖТ  const enumPositions = new Map<number, string>(); // position тЖТ enum type name
   828тЖТ   828тЖТ  columns.forEach((col, idx) => {
   829тЖТ   829тЖТ    const enumType = ENUM_TYPE_MAP[`${tableName}.${col}`];
   830тЖТ   830тЖТ    if (enumType) {
   831тЖТ   831тЖТ      enumPositions.set(idx, enumType);
   832тЖТ   832тЖТ    }
   833тЖТ   833тЖТ  });
   834тЖТ   834тЖТ
   835тЖТ   835тЖТ  if (enumPositions.size === 0) return sql;
   836тЖТ   836тЖТ
   837тЖТ   837тЖТ  // Parse the VALUES clause тАФ tokenize respecting strings and nested parens
   838тЖТ   838тЖТ  const values: string[] = [];
   839тЖТ   839тЖТ  let currentValue = '';
   840тЖТ   840тЖТ  let inStr = false;
   841тЖТ   841тЖТ  let strChar = '';
   842тЖТ   842тЖТ  let depth = 0;
   843тЖТ   843тЖТ
   844тЖТ   844тЖТ  for (let i = 0; i < valuesStr.length; i++) {
   845тЖТ   845тЖТ    const ch = valuesStr[i];
   846тЖТ   846тЖТ
   847тЖТ   847тЖТ    if (!inStr && (ch === "'" || ch === '"')) {
   848тЖТ   848тЖТ      inStr = true;
   849тЖТ   849тЖТ      strChar = ch;
   850тЖТ   850тЖТ      currentValue += ch;
   851тЖТ   851тЖТ      continue;
   852тЖТ   852тЖТ    }
   853тЖТ   853тЖТ
   854тЖТ   854тЖТ    if (inStr) {
   855тЖТ   855тЖТ      currentValue += ch;
   856тЖТ   856тЖТ      if (ch === strChar) {
   857тЖТ   857тЖТ        if (valuesStr[i + 1] === strChar) {
   858тЖТ   858тЖТ          currentValue += valuesStr[i + 1];
   859тЖТ   859тЖТ          i++;
   860тЖТ   860тЖТ          continue;
   861тЖТ   861тЖТ        }
   862тЖТ   862тЖТ        inStr = false;
   863тЖТ   863тЖТ      }
   864тЖТ   864тЖТ      continue;
   865тЖТ   865тЖТ    }
   866тЖТ   866тЖТ
   867тЖТ   867тЖТ    if (ch === '(') { depth++; currentValue += ch; continue; }
   868тЖТ   868тЖТ    if (ch === ')') { depth--; currentValue += ch; continue; }
   869тЖТ   869тЖТ
   870тЖТ   870тЖТ    if (ch === ',' && depth === 0) {
   871тЖТ   871тЖТ      values.push(currentValue.trim());
   872тЖТ   872тЖТ      currentValue = '';
   873тЖТ   873тЖТ      continue;
   874тЖТ   874тЖТ    }
   875тЖТ   875тЖТ
   876тЖТ   876тЖТ    currentValue += ch;
   877тЖТ   877тЖТ  }
   878тЖТ   878тЖТ  if (currentValue.trim()) values.push(currentValue.trim());
   879тЖТ   879тЖТ
   880тЖТ   880тЖТ  // Add enum casts to parameters at enum column positions
   881тЖТ   881тЖТ  let changed = false;
   882тЖТ   882тЖТ  for (let i = 0; i < columns.length && i < values.length; i++) {
   883тЖТ   883тЖТ    const enumType = enumPositions.get(i);
   884тЖТ   884тЖТ    if (enumType) {
   885тЖТ   885тЖТ      // Match $N parameter and add cast
   886тЖТ   886тЖТ      const paramMatch = values[i].match(/^(\$\d+)$/);
   887тЖТ   887тЖТ      if (paramMatch) {
   888тЖТ   888тЖТ        values[i] = `${paramMatch[1]}::"${enumType}"`;
   889тЖТ   889тЖТ        changed = true;
   890тЖТ   890тЖТ      }
   891тЖТ   891тЖТ    }
   892тЖТ   892тЖТ  }
   893тЖТ   893тЖТ
   894тЖТ   894тЖТ  if (!changed) return sql;
   895тЖТ   895тЖТ
   896тЖТ   896тЖТ  return `${prefix}${columnsStr}${middle}${values.join(', ')}${suffix}`;
   897тЖТ   897тЖТ}

   916тЖТ   916тЖТ/**
   917тЖТ   917тЖТ * Normalize a boolean value from either SQLite (0/1) or PostgreSQL (true/false).
   918тЖТ   918тЖТ * Use this in API route handlers to safely check boolean columns from result rows.
   919тЖТ   919тЖТ *
   920тЖТ   920тЖТ * @example
   921тЖТ   921тЖТ * // Works with both SQLite (returns 0/1) and PostgreSQL (returns true/false)
   922тЖТ   922тЖТ * const isActive = normalizeBool(row.isActive);
   923тЖТ   923тЖТ * if (isActive) { ... }
   924тЖТ   924тЖТ */
   925тЖТ   925тЖТexport function normalizeBool(value: unknown): boolean {
   926тЖТ   926тЖТ  if (typeof value === 'boolean') return value;
   927тЖТ   927тЖТ  if (value === 1 || value === '1' || String(value) === 'true') return true;
   928тЖТ   928тЖТ  return false;
   929тЖТ   929тЖТ}
   930тЖТ   930тЖТ
   931тЖТ   931тЖТ/**
   932тЖТ   932тЖТ * Normalize row values: convert PostgreSQL booleans to JavaScript booleans
   933тЖТ   933тЖТ * so that API route handlers always see consistent types regardless of backend.
   934тЖТ   934тЖТ *
   935тЖТ   935тЖТ * SQLite returns: 0, 1 (number) for booleans
   936тЖТ   936тЖТ * PostgreSQL returns: false, true (boolean) for booleans
   937тЖТ   937тЖТ *
   938тЖТ   938тЖТ * We normalize BOTH to JavaScript boolean, so route handlers can always use
   939тЖТ   939тЖТ * `normalizeBool(row.col)` or `if (row.col)` consistently.
   940тЖТ   940тЖТ */
   941тЖТ   941тЖТfunction normalizeRow(row: Record<string, unknown>): Record<string, unknown> {
   942тЖТ   942тЖТ  const normalized: Record<string, unknown> = {};
   943тЖТ   943тЖТ  const boolColSet = new Set(BOOLEAN_COLUMNS);
   944тЖТ   944тЖТ
   945тЖТ   945тЖТ  for (const [key, value] of Object.entries(row)) {
   946тЖТ   946тЖТ    // Strip quotes from quoted identifiers (PostgreSQL returns "ColName" тЖТ ColName)
   947тЖТ   947тЖТ    const cleanKey = key.replace(/^"(.*)"$/, '$1');
   948тЖТ   948тЖТ
   949тЖТ   949тЖТ    if (typeof value === 'bigint') {
   950тЖТ   950тЖТ      normalized[cleanKey] = Number(value);
   951тЖТ   951тЖТ    } else if (boolColSet.has(cleanKey)) {
   952тЖТ   952тЖТ      normalized[cleanKey] = normalizeBool(value);
   953тЖТ   953тЖТ    } else {
   954тЖТ   954тЖТ      normalized[cleanKey] = value;
   955тЖТ   955тЖТ    }
   956тЖТ   956тЖТ  }
   957тЖТ   957тЖТ  return normalized;
   958тЖТ   958тЖТ}

   960тЖТ   960тЖТ/**
   961тЖТ   961тЖТ * Convert SQLite-compatible argument values to PostgreSQL-compatible types.
   962тЖТ   962тЖТ *
   963тЖТ   963тЖТ * PostgreSQL is strict about types:
   964тЖТ   964тЖТ * - BOOLEAN columns need true/false, not 0/1
   965тЖТ   965тЖТ * - TIMESTAMP columns need Date objects, not ISO strings
   966тЖТ   966тЖТ *
   967тЖТ   967тЖТ * This function scans the SQL for boolean column assignment patterns (e.g., `availableNow = ?`)
   968тЖТ   968тЖТ * and converts the corresponding argument from 0/1 to false/true.
   969тЖТ   969тЖТ * It also converts ISO date strings to Date objects.
   970тЖТ   970тЖТ */
   971тЖТ   971тЖТfunction convertArgsForPostgres(
   972тЖТ   972тЖТ  rawSql: string,
   973тЖТ   973тЖТ  args: (string | number | boolean | null | Buffer | unknown)[]
   974тЖТ   974тЖТ): (string | number | boolean | null | Date | unknown)[] {
   975тЖТ   975тЖТ  // Detect boolean column positions by scanning SQL for `colName = ?` patterns
   976тЖТ   976тЖТ  const boolQMarkIndices = new Set<number>();
   977тЖТ   977тЖТ  const boolColRegex = new RegExp(
   978тЖТ   978тЖТ    `\\b(${BOOLEAN_COLUMNS_PATTERN})\\s*=\\s*\\?`,
   979тЖТ   979тЖТ    'gi'
   980тЖТ   980тЖТ  );
   981тЖТ   981тЖТ  let boolMatch;
   982тЖТ   982тЖТ  while ((boolMatch = boolColRegex.exec(rawSql)) !== null) {
   983тЖТ   983тЖТ    const beforeSql = rawSql.substring(0, boolMatch.index + boolMatch[0].length);
   984тЖТ   984тЖТ    const qCount = (beforeSql.match(/\?/g) || []).length;
   985тЖТ   985тЖТ    boolQMarkIndices.add(qCount - 1); // 0-indexed
   986тЖТ   986тЖТ  }
   987тЖТ   987тЖТ
   988тЖТ   988тЖТ  // Detect LIMIT/OFFSET positions тАФ PostgreSQL requires integer types, not strings
   989тЖТ   989тЖТ  const limitOffsetIndices = new Set<number>();
   990тЖТ   990тЖТ  const limitOffsetRegex = /\bLIMIT\s+\?|\bOFFSET\s+\?/gi;
   991тЖТ   991тЖТ  let loMatch;
   992тЖТ   992тЖТ  while ((loMatch = limitOffsetRegex.exec(rawSql)) !== null) {
   993тЖТ   993тЖТ    const beforeSql = rawSql.substring(0, loMatch.index + loMatch[0].length);
   994тЖТ   994тЖТ    const qCount = (beforeSql.match(/\?/g) || []).length;
   995тЖТ   995тЖТ    limitOffsetIndices.add(qCount - 1); // 0-indexed
   996тЖТ   996тЖТ  }
   997тЖТ   997тЖТ
   998тЖТ   998тЖТ  return args.map((arg, i) => {
   999тЖТ   999тЖТ    if (arg === undefined) return null;
  1000тЖТ  1000тЖТ    // Convert 0/1 integers to booleans for boolean column positions
  1001тЖТ  1001тЖТ    if (boolQMarkIndices.has(i) && (arg === 0 || arg === 1)) {
  1002тЖТ  1002тЖТ      return arg === 1;
  1003тЖТ  1003тЖТ    }
