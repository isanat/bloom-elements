   179→            experienceRating,

   207→    }

   215→        method: "PATCH",
   216→        headers: { "Content-Type": "application/json" },

   218→          questionnaire: {
   219→            familyRating: cgFamilyRating,
   220→            clarityRating: cgClarityRating,
   221→            notes: cgNotes,
   222→            interviewConductedProperly: cgInterviewConducted,

   254→      return url;

   261→    onChange,
   262→    label,

   264→    value: number;
   265→    onChange: (v: number) => void;
   266→    label: string;

   303→      <PageLayout spacing="normal">
   304→        <div className="space-y-4 max-w-md mx-auto">

   306→            variant="error"
   307→            title="Erro"

   311→            className="rounded-xl h-12 w-full"

   380→            </div>

   385→        {error && (
   386→          <BloomAlert
   387→            variant="error"
   388→            title="Erro"
   389→            message={error}
   390→          />
   391→        )}
   392→
   393→        {/* ====== SCHEDULED ====== */}
   394→        {interview.status === "SCHEDULED" && (
   395→          <div className="space-y-5">
   396→            <BloomCard>

   409→                  {
   410→                    icon: "\uD83D\uDD07",
   411→                    text: "Escolha um local silencioso e bem iluminado",
   412→                  },
   413→                  {

   417→                  {

   428→                    </span>
   429→                    <span>{tip.text}</span>

   440→                <IconVideo className="h-5 w-5 mr-2" />
   441→                Entrar na Sala de Entrevista
   442→              </Button>

   444→                variant="outline"
   445→                onClick={() => router.push("/app/interviews")}
   446→                size="lg"
   447→                className="flex-1 h-12"
   448→              >
   449→                <IconArrowLeft className="h-4 w-4 mr-2" />
   450→                Voltar às Entrevistas
   451→              </Button>
   452→            </div>
   453→          </div>

   455→
   456→        {/* ====== IN PROGRESS ====== */}

   467→            {/* Warning when user left meeting - Bloom Elements */}

   469→              <BloomAlert
   470→                variant="warning"
   471→                title="Saiu da reunião"

   473→              />

   488→                size="lg"

   495→            </div>
   496→            <Button

   498→              size="sm"
   499→              onClick={() => router.push("/app/interviews")}

   502→              <IconArrowLeft className="h-4 w-4 mr-2" />

   523→              {/* Section 1: About the interview itself */}

   529→                  <StarRating
   530→                    label="Comunicação"
   531→                    value={communicationRating}

   577→                      <span className="text-2xl block mb-2">
